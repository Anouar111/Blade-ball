-- Fonction sécurisée pour charger les scripts
local function safeLoad(name, url)
    print("Tentative de chargement : " .. name)
    local success, content = pcall(function()
        return game:HttpGet(url)
    end)

    if success and content then
        local func, err = loadstring(content)
        if func then
            task.spawn(function()
                local runSuccess, runErr = pcall(func)
                if not runSuccess then
                    warn(name .. " Erreur d'exécution : " .. tostring(runErr))
                end
            end)
            print(name .. " chargé avec succès.")
        else
            warn(name .. " Erreur de compilation : " .. tostring(err))
        end
    else
        warn("Impossible de récupérer le lien pour " .. name)
    end
end
