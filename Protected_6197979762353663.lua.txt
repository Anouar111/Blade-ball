-- Fonction de chargement sécurisée
local function safeLoad(url, name)
    local success, content = pcall(function()
        return game:HttpGet(url)
    end)

    if success and content and #content > 0 then
        local execSuccess, err = pcall(function()
            loadstring(content)()
        end)
        if not execSuccess then
            warn("Erreur d'exécution [" .. name .. "]: " .. tostring(err))
        end
    else
        warn("Erreur de téléchargement [" .. name .. "]: Impossible de récupérer l'URL")
    end
end

-- 1. Lancement de Blade Ball (Argon)
task.spawn(function()
    print("Chargement de Argon...")
    safeLoad("https://4x.wtf/loader", "Argon")
end)

-- 2. Lancement de EBK avec un délai plus sûr
task.spawn(function()
    print("Attente pour EBK...")
    task.wait(8) -- Augmenté à 8 secondes pour laisser le temps au GUI de s'initialiser
    print("Chargement de EBK...")
    safeLoad("https://pastefy.app/587JqZsk/raw", "EBK")
end)
